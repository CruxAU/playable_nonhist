add_namespace = mtg_germany

# Event for HOL demanding they join GER's faction and wars or die
country_event = {
	id = mtg_germany.1
	title = mtg_germany.1.t
	desc = mtg_germany.1.d
	picture = GFX_report_event_german_troops

	is_triggered_only = yes
	
	# Join faction
	option = {
		name = mtg_germany.1.a
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				NOT = {
					has_completed_focus = HOL_cave_to_the_germans
				}
			}
		}
		hidden_effect = { set_country_flag = { flag = HOL_joined_germany_flag value = 1 days = 730 } }
		effect_tooltip = {
			FROM = { add_to_faction = ROOT }
		}
		hidden_effect = {
			add_ai_strategy = {
				type = alliance
				id = FROM
				value = 200
			}
		}
		FROM = {
			hidden_effect = {
				add_ai_strategy = {
					type = alliance
					id = ROOT
					value = 200
				}
			}
			country_event = {
				id = mtg_germany.2
			}
		}
	}

	# Create wargoal
	option = {
		name = mtg_germany.1.b
		ai_chance = {
			factor = 1
		}
		effect_tooltip = {
			FROM = {
				create_wargoal = {
					type = take_state_focus
					target = HOL
					generator = { 7 }
				}
			}
		}
		FROM = {
			country_event = {
				id = mtg_germany.3
			}
		}
	}
}
